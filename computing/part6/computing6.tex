unsigned i2f(int i) {
  if (i == 0) return 0;
  int abs = i < 0 ? -i : i;

  int s = i > 0 ? 0b0 : 0x80000000;

  int tmp = abs;
  int right_shifts = 0;
  while (0b01 != abs) {
    abs = abs >> 1;
    right_shifts++;
  }
  int e = (127 + right_shifts) << 23;

  tmp = (tmp << (32 - right_shifts)) >> 9;
  int m = tmp & 0x007FFFFF;

  return s | e | m;
}
\end{lstlisting}
